CREATE TABLE yytd (
    DT DATE,
    省 VARCHAR(255),
    市 VARCHAR(255),
    区县 VARCHAR(255),
    土地出让收入 FLOAT
);

#按照新的行政区划重新统计
INSERT INTO yytd
SELECT 
A.`Report Date` AS DT, B.省, B.市, B.区县,
SUM(土地出让收入) AS 土地出让收入
FROM `yyqy` A
INNER JOIN xzqh_yy B
ON A.`Administrative Division Name`=B.区域
GROUP BY A.`Report Date`, B.省, B.市, B.区县;


#更新市级没数据的
CREATE TEMPORARY TABLE temp_table AS
SELECT DT, 省, 市, SUM(土地出让收入) as sum_土地出让收入
FROM yytd
WHERE 区县 != '' AND 省 != ''
GROUP BY DT, 省, 市;

UPDATE yytd
SET 土地出让收入 = (
  SELECT sum_土地出让收入
  FROM temp_table
  WHERE temp_table.DT = yytd.DT AND temp_table.省 = yytd.省 AND temp_table.市 = yytd.市
)
WHERE 市 != '' AND 市 != '省本级' AND 区县 = '' AND 土地出让收入 is null;

DROP TEMPORARY TABLE temp_table;

#更新、补充市本级
CREATE TEMPORARY TABLE temp_table AS
SELECT DT, 省, 市, '市本级' as 区县, SUM(case when 区县='' then 土地出让收入 else 0 end)-SUM(case when 区县!='市本级' and 区县!='' then 土地出让收入 else 0 end) as 土地出让收入
FROM yytd
WHERE 省 != '' and 市!='' and 省 not in ('北京市','上海市','重庆市','天津市') and 市 != '省直辖县级行政区划'
GROUP BY DT, 省, 市;

UPDATE yytd
SET 土地出让收入 = (
  SELECT 土地出让收入
  FROM temp_table
  WHERE temp_table.DT = yytd.DT AND temp_table.省 = yytd.省 AND temp_table.市 = yytd.市 AND temp_table.区县 = yytd.区县
)
WHERE 土地出让收入 is null;

DROP TEMPORARY TABLE temp_table;
