    global _cursor_bond
    while retry_count < max_retries:
        try:
            # 开始事务
            trans_bond = _cursor_bond.begin()
            try:
                _cursor_bond.execute(text(sql1))
                # 提交事务
                trans_bond.commit()

                # 如果出错，回滚事务

def split_column(df, column):
    if df is None or column not in df:
        return pd.DataFrame()
    temp_df = df[column].str.split(';', expand=True).stack()
    temp_df.index = temp_df.index.droplevel(-1) 
    temp_df = temp_df.str.extract(r'\((.*?)\) (\d+\.\d+)', expand=True)
    temp_df.columns = ['dt', 'value']
    temp_df['指标']=column
    temp_df['trade_code'] = df['thscode'].iloc[0]
    
    return temp_df
